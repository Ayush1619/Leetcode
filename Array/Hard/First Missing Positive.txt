class Solution {
public:
    int firstMissingPositive(vector<int>& A) {
      //O(N) SPACE=O(1) 
        // sort(A.begin(),A.end());
        // int diff=0;
        for(int i=0;i<A.size();i++)
            while(A[i]>0 && A[i]<=A.size() && A[A[i]-1]!=A[i])
                swap(A[i],A[A[i]-1]);
        
        for(int i=0;i<A.size();i++)
            if(A[i]!=i+1)
                return i+1;
        
        return A.size()+1;
    }
};
