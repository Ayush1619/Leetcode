class Solution {
public:
    vector<int> findDisappearedNumbers(vector<int>& nums) {
        int n = nums.size();

        for(int i=0;i<nums.size();i++){
             int current = abs(nums[i]);
            nums[current-1] = -abs(nums[current-1]);
            
          }
        vector<int> result;
        for(int i=0;i<nums.size();i++){
            if(nums[i]>0){
                result.push_back(i+1);
            }
        }
        
        return result;
    }
};
