class Solution {
public:
    int maxSubArray(vector<int>& nums) {
        int mx_sum=INT_MIN;
        int count=0;
        int n=nums.size();
       
        for(int i=0;i<n;i++){
            count+=nums[i];
            if(count>mx_sum){
                mx_sum=count;
            }
            if(count<0){
                count=0;
               }
        }
        return mx_sum;
    }
};
