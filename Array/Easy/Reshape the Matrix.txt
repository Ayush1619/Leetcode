class Solution {
public:
    vector<vector<int>> matrixReshape(vector<vector<int>>& nums, int r, int c) {
        int n=nums.size(),col=nums[0].size();
        if(n*col!=r*c)
            return nums;
        vector<vector<int>> ans(r,vector<int>(c,0));
        for(int i=0;i<r*c;i++)
            ans[i/c][i%c]=nums[i/col][i%col];
        return ans;
    }
};
