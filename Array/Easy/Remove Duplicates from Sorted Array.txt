class Solution {
public:
    int removeDuplicates(vector<int>& nums) {
        vector<int>v;
        map<int,int>mp;
        for(int i=0;i<nums.size();i++){
            mp[nums[i]]++;
        }
        for(auto it=mp.begin();it!=mp.end();it++){
            v.push_back(it->first);
                
        }
        nums=v;
        return nums.size();
        
    }
};
