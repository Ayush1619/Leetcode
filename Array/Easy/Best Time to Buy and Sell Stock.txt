class Solution {
public:
    int maxProfit(vector<int>& prices) {
        int mn=INT_MAX;
        int count=0;
        for(int i=0;i<prices.size();i++){
            // for(int j=i+1;j<prices.size();j++){
                // int profit=prices[j]-prices[i];
                if(prices[i]<mn)
                    mn=prices[i];
            else if(prices[i]-mn>count)
                count=prices[i]-mn;
            // }
        }
        return count;
        
    }
};
