class Solution {
public:
    int findMaxConsecutiveOnes(vector<int>& nums) {
        int n = nums.size();
        int sum=0;
        int result=0;
        for(int i=0;i<nums.size();i++){
            if(nums[i] == 1){
               sum++;
            }
            else{
                sum=0;
            }
            result = max(sum,result);
        }
        return result;
    }
};
