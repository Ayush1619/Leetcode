class Solution {
public:
    int Findnumbers(vector<int>& nums) {
        int count=0;
        for(int i=0;i<nums.size();i++){
          
            for(int j=1;j<nums.size();j++)
                if((nums[i] + nums[j]) % 2 == 0){
                    count++;
                    continue;
                }
            else{
                return 0;
            }
        }
        return count;
    }
};
