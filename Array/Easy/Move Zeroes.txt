class Solution {
public:
    void moveZeroes(vector<int>& nums) {
        // if(nums.size()==1 && nums[0]==0)
        //     return [0];
        int l=0;
         for(int i=0;i<nums.size();i++){
             if(nums[i]!=0)
                 nums[l++]=nums[i];
         }
         for(int i=l;i<nums.size();i++){
             nums[i]=0;
         }
    }
};
