class Solution {
public:
    bool canPlaceFlowers(vector<int>& flowerbed, int n) {
        int x = flowerbed.size();
        int count=0;
        for(int i=0;i<flowerbed.size();i++){
            if(flowerbed[i] ==0){
               if((i==0 || flowerbed[i-1]==0) && (i==x-1 || flowerbed[i+1]==0)){
                 
                   flowerbed[i]=1;
                    count++; 
               }
                
            }
            
        }
        return count>=n;
    }
};
