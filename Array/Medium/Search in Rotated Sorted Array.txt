class Solution {
public:
    int search(vector<int>& nums, int target) {
        int r=nums.size()-1;
        int l=0;
    
        while(l<=r){
            if(nums[l]==target){
                return l;

            }
            else if(nums[r]==target){
                return r;
            }
            else{
                l++;
                r--;
            
            }
                
            
                }
        
        
        return -1;
    }
};
