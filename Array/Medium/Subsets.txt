class Solution {
public:
     vector<vector<int>> guddu; //result
        vector<int> munna; //temp
    vector<vector<int>> subsets(vector<int>& nums) {
        int n = nums.size();
       
        kaleen(0,n,nums);
         return guddu;
    }
    
    void kaleen(int index,int n, vector<int>nums){
        
        if(index>=n){
            if(index == n)
                guddu.push_back(munna);
            return;
        }
    
        kaleen(index+1,n,nums);
            munna.push_back(nums[index]);
        kaleen(index+1,n,nums);
            munna.pop_back();
      
    }
};
