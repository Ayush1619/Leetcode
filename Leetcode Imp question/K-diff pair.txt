using namespace std;

class Solution {
public:
    int findPairs(vector<int>& nums, int k) {
        
        sort(nums.begin(),nums.end());
        int ans = 0;
        int n = nums.size();
        int l = 0, r = 0;
        
        for (l; l < n; l++) {
            for (r = max(l + 1, r); r < n && nums[r] - nums[l] < k; r++) ;
            if (r < n && nums[r] - nums[l] == k) ans++;
            while (l + 1 < n && nums[l] == nums[l + 1]) l++;
        }
        return ans;
    }
};
