class Solution {
public:
    bool isToeplitzMatrix(vector<vector<int>>& matrix) {
        bool toeplitz=true;
        int n=matrix.size();
        for(int i=0;i<n;i++){
            for(int j=1;j<n;j++){
               
                if(matrix[i++]==matrix[j++]){
                    toeplitz=false;
                }
            }
        }
        return toeplitz;
    }
};
