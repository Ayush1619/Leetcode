class Solution {
public:
    bool search(vector<int>& nums, int target) {
        //nums is given after rotation,target
        //return true if target is present in nums ,or false otherwise
        int n=nums.size();
        bool search=false;
        for(int i=0;i<n;i++){
            if(nums[i]==target){
                search=true;
            }
            else{
                false;
            }
        }
        return search;
    }
};
