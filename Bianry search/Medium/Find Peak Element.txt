class Solution {
public:
    int findPeakElement(vector<int>& nums) {
        // sort(nums.begin(),nums.end());
        int n=nums.size();
        // int count=0;
        int i;
        for(int i=0;i<n-1;i++){
            if(nums[i]>nums[i+1])
                return i;
        
        }
        return n-1;
        
    }
};
