class Solution {
public:
    bool find132pattern(vector<int>& nums) {
        // bool pat=true;
        int n=nums.size();
        stack<int>st;
        int x=INT_MIN;
        
        for(int i=n-1;i>=0;i--){
            if(x>nums[i])
                return true;
            
        
        while(!st.empty() && nums[i]>st.top()){
            x=st.top();
            st.pop();
        }
            st.push(nums[i]);
        }
       return false;
    }
};
