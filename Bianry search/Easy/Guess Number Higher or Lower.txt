/** 
 * Forward declaration of guess API.
 * @param  num   your guess
 * @return 	     -1 if num is lower than the guess number
 *			      1 if num is higher than the guess number
 *               otherwise return 0
 * int guess(int num);
 */

class Solution {
public:
    int guessNumber(int n) {
      int l=0;
        int h=n;
        int mid=0;
        int ans=0;
        while(l<=h){
           mid=l+(h-l)/2;
            ans=guess(mid);
            if(ans==0)
                return mid;
            else if(ans<0)
                h=mid-1;
            else
                l=mid+1;
        }
      return -1;
       
    }
};
