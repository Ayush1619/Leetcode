class Solution {
public:
    char nextGreatestLetter(vector<char>& letters, char target) {
      int l=0;
      int r=letters.size();
       
        while(l<r){
           int mid=l+(r-l)/2;
            if(target>=letters[mid])
               l=mid+1;
            else
                r=mid;
        }
        return letters[l%letters.size()];
    }
};
