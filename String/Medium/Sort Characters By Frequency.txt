class Solution {
public:
string frequencySort(string s) {
string ans="";
map<char,int> m;
for(int i=0;i<s.length();i++)
{
m[s[i]]++;
}
priority_queue<pair<int,char> >pq;
for(auto it:m)
{
pq.push({it.second,it.first});
}
while(!pq.empty())
{
int a=pq.top().first;char c=pq.top().second;
for(int i=0;i<a;i++)
ans+=c;
pq.pop();
}
return ans;
}
};
