class Solution {
public:
    int minimumSwap(string s1, string s2) 
    {
        int x1=0;
        int y1=0;
        for(int i=0;i<s1.size();i++)
        {
            if(s1[i]=='x'&&s2[i]=='y')
            {
                x1++;
            }
            if(s1[i]=='y'&&s2[i]=='x')
            {
                y1++;
            }
        }
        if((x1+y1)%2==1)
        {
            return -1;
        }
        return ceil(double(x1/2.0))+ceil(double(y1/2.0));
    }
};
