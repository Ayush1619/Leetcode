class Solution {
public:
    string countAndSay(int n) {
        if (n == 1) return "1"; // edge case 
        
        string s = countAndSay(n-1), ans = "";
        int cnt = 0; 
        for (int i = 0; i <= s.size(); ++i) {
            if (i == s.size() or (i > 0 and s[i-1] != s[i])) {
                ans += to_string(cnt) + s[i-1]; 
                cnt = 0; 
            }
            cnt += 1; 
        }
        return ans; 
    }
};
