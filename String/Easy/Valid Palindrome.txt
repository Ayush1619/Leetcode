class Solution {
public:
    bool palin(string & dup)
    {
        int s= dup.size();
        for(int i=0;i<s/2;i++)
        {
            if(dup[i]!=dup[s-1-i]) return false;
        }
        return true;
    }
    bool isPalindrome(string s) {
        string dup="";
        for(auto c: s)
        {
            if(isalpha(c)||isdigit(c)) dup+=tolower(c);
        }
        return palin(dup);
    }
};
