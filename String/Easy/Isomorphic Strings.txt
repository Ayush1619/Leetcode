class Solution {
public:
    bool isIsomorphic(string s, string t) {
        if(s.size()!=t.size()){
            return false;
        }
        unordered_map<char,int>mp;
        unordered_map<char,int>mp1;
        for(int i=0;i<s.size();i++){
            mp[s[i]]=t[i];
            mp1[t[i]]=s[i];
        }
        for(int i=0;i<s.size();i++){
            if(mp[s[i]]!=t[i] || mp1[t[i]]!=s[i]){
                return false;
            }
        }
        return true;
    }
};
